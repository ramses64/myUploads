private async Task OnNativeDropAsync(string[] paths, Point screenPt)
{
    // Ensure UI thread (WinForms equivalent of Dispatcher)
    if (InvokeRequired)
    {
        await (Task)Invoke(new Func<Task>(() => OnNativeDropAsync(paths, screenPt)));
        return;
    }

    var clientPt = this.PointToClient(screenPt);

    Control? hit = DeepHitTest(this, clientPt);
    if (hit == null) return;

    var handler = FindDropHandler(hit);
    if (handler == null) return;

    var container = await DragDropHandler.CreateFromFilePaths(paths);
    await handler.HandleNativeDropAsync(container);
}
