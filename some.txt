private static INativeDropHandler? FindDropHandler(Control? start)
{
    Control? current = start;
    while (current != null)
    {
        // If the control itself implements the handler
        if (current is INativeDropHandler h)
            return h;

        // Optional: if you attach a handler object via Tag
        if (current.Tag is INativeDropHandler tagHandler)
            return tagHandler;

        current = current.Parent;
    }
    return null;
}
