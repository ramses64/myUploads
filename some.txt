private UIElement? FindDropTarget(DependencyObject start)
{
    DependencyObject? cur = start;
    while (cur != null)
    {
        // You can refine this check:
        // - UIElement has AllowDrop property
        // - Some controls handle Drop even if AllowDrop isn't set (repo-dependent)
        if (cur is UIElement el && el.AllowDrop)
            return el;

        cur = System.Windows.Media.VisualTreeHelper.GetParent(cur);
    }
    return null;
}
